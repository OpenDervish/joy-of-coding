#include <stdio.h>
#include <string.h>
int main()
{
    int i, index=0;
    char string[50], abbr[10];

    printf("Enter a string: ");
    scanf("%[^\n]%*c", string);

    abbr[index]=string[0];
    index++;

    for(i=0; i<strlen(string); i++)
    {
        if(string[i]==' ')
        {
            abbr[index]=string[i+1];
            index++;
        }
    }

    abbr[index]='\0';

    printf("Abbreviating \"%s\": %s", string, abbr);
    printf("\n");

    return 0;
}

#include <stdio.h>
#include <string.h>
int main()
{
    int i, count=1;
    char string[30];

    printf("Enter a string: ");
    scanf("%[^\n]%*c", string);

    for(i=0; i<strlen(string); i++)
    {
        if(string[i]==' ')
            count++;
    }
    string[i]='\0';
    printf("%s has %d words.", string, count);
    printf("\n");

    return 0;
}
